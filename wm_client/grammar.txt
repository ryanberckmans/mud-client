
start
  = cmd:cmd ";" start:start { return cmd.concat(start); }
  / cmd

cmd
  = "/" i:integer text:text { var cmds = []; for(;i>0;i--) cmds.push(text); return cmds; }
  / text:text { return [text]; }

integer
  = digits:[0-9]+ { return parseInt(digits.join(""), 10); }

text 
  = s:[^;]* { return s.join(""); }
